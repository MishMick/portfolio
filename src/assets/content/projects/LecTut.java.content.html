/**
 * LecTut is the official study portal of IIT Roorkee.
 * Students and professors upload study material
 * such as lecture notes, classroom slide decks and quizzes.
 *
 * @since {} February 2017
 * @deprecated {} April 2017
 */
public class LecTut {

  String[] attributes;
  HashMap<String, String> links;

  /**
   * Mandatory default constructor
   */
  public LecTut() {
    // Do nothing
  }

  /**
   * Return a new instance of the app.
   *
   * @return newly created LecTut instance
   */
  public static LecTut newInstance () {
    LecTut app = new LecTut();

    app.attributes = {
      Developed a companion Android app for LecTut.,
      Built offline capabilities into the app.,
      Allows study from downloaded sources.
    }
    app.links.put(
      Play store,
      https://play.google.com/store/apps/details?id=in.channeli.lectut
    )

    return app;
  }
}
